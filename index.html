let currentIndex = 0; let widget = null; let isShuffle = false; let isRepeat = false; // Initialize and bind the player function loadAlbum(index, autoPlay = false) { currentIndex = index; const album = playlists[index]; const iframe = document.getElementById('soundcloud-player'); // Construct the standard widget URL embedding options iframe.src = `https://soundcloud.com{encodeURIComponent(album.url)}&color=%23ffcc00&auto_play=${autoPlay}&visual=true&show_comments=true`; // Bind to the new iframe context using the SoundCloud API library widget = SC.Widget(iframe); // Attach listeners once the widget is ready widget.bind(SC.Widget.Events.READY, () => { setupPlayerListeners(); }); updateNav(); } // Dynamically generate navigation elements and ensure active is centered function updateNav() { const nav = document.getElementById('album-nav'); nav.innerHTML = ''; playlists.forEach((album, i) => { const btn = document.createElement('button'); btn.className = 'nav-btn'; btn.textContent = album.name; btn.onclick = () => loadAlbum(i, true); // Auto-play if user manually clicks a collection if (i === currentIndex) { btn.classList.add('active'); btn.scrollIntoView({ behavior: 'smooth', block: 'nearest', inline: 'center' }); } nav.appendChild(btn); }); } // Core hardware control simulations function togglePlay() { if (widget) widget.toggle(); } function prev() { if (widget) widget.prev(); } function next() { if (widget) widget.next(); } // Advanced custom playback toggles function toggleShuffle() { isShuffle = !isShuffle; const btn = document.querySelector("button[onclick='toggleShuffle()']"); btn.style.background = isShuffle ? "#ffcc00" : "transparent"; btn.style.color = isShuffle ? "#000" : "#ffcc00"; } function toggleRepeat() { isRepeat = !isRepeat; const btn = document.querySelector("button[onclick='toggleRepeat()']"); btn.style.background = isRepeat ? "#ffcc00" : "transparent"; btn.style.color = isRepeat ? "#000" : "#ffcc00"; } // Action Bar API Integrations function shareCurrent() { if (!widget) return; widget.getCurrentSound((sound) => { const shareUrl = sound ? sound.permalink_url : window.location.href; navigator.clipboard.writeText(shareUrl) .then(() => alert('Track link copied to clipboard!')) .catch(() => alert('Failed to copy link. Here it is: ' + shareUrl)); }); } // Logic listening framework to determine next items when a song concludes function setupPlayerListeners() { // Dynamic UI matching for the custom Play/Pause trigger button widget.bind(SC.Widget.Events.PLAY, () => { document.querySelector(".controls .btn:nth-child(2)").textContent = "⏸️"; }); widget.bind(SC.Widget.Events.PAUSE, () => { document.querySelector(".controls .btn:nth-child(2)").textContent = "▶️"; }); // Intelligent queue handling upon target track completion widget.bind(SC.Widget.Events.FINISH, () => { if (isRepeat) { widget.seekTo(0); widget.play(); } else if (isShuffle) { // Pick a completely random track inside the current collection context widget.getSounds((sounds) => { if (sounds && sounds.length > 1) { const randomIndex = Math.floor(Math.random() * sounds.length); widget.skip(randomIndex); } else { autoAdvanceAlbum(); } }); } else { // Standard linear logic: assess if there is a next track in the playlist widget.getSounds((sounds) => { widget.getCurrentSoundIndex((currentIndexInSoundcloud) => { if (currentIndexInSoundcloud === sounds.length - 1) { // End of the current album achieved -> wrap to next album autoAdvanceAlbum(); } else { widget.next(); } }); }); } }); } // Cycle naturally down to the next playlist entry when one finishes completely function autoAdvanceAlbum() { let nextIndex = currentIndex + 1; if (nextIndex >= playlists.length) { nextIndex = 0; // Wrap back around to the first album if index overshot } loadAlbum(nextIndex, true); } // Kickstart deployment pipeline window.onload = () => { if (playlists.length > 0) loadAlbum(0, false); };
